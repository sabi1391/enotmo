import boto3
import os
os.environ['AWS_PROFILE'] = "jenkinsmum"
AWS_region = 'ap-south-1'
EC2_RESOURCE = boto3.resource('ec2', region_name=AWS_region)
INSTANCE_ID = ['i-0fe476d0bc705dcf0', 'i-036eb6471e4c08d08', 'i-0e207ffae067a0a48']
for value in INSTANCE_ID:
    instance = EC2_RESOURCE.Instance(value)
    instance.start()
    print('Starting EC2 instance: %s' %  (instance.id))
    instance.wait_until_running()
    print('EC2 instance has been started: %s' %  (instance.id))
